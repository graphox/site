315:    * Produces a compliant, formatted RFC 2822 'phrase' based on the string given.
316:    * @param Swift_Mime_Header $header
317:    * @param string $string as displayed
318:    * @param string $charset of the text

321:    * @return string

324:     Swift_Mime_HeaderEncoder $encoder = null, $shorten = false)
325:   {
326:     //Treat token as exactly what was given
327:     $phraseStr = $string;

336:           $phraseStr, array('"'), $this->_specials

564:   private function _tokensToString(array $tokens)

566:     $lineCount = 0;
567:     $headerLines = array();
568:     $headerLines[] = $this->_name . ': ';
569:     $currentLine =& $headerLines[$lineCount++];
570:     
571:     //Build all tokens back into compliant header
572:     foreach ($tokens as $i => $token)
573:     {
574:       //Line longer than specified maximum or token was just a new line
575:       if (("\r\n" == $token) ||
576:         ($i > 0 && strlen($currentLine . $token) > $this->_lineLength)
577:         && 0 < strlen($currentLine))
578:       {
579:         $headerLines[] = '';
580:         $currentLine =& $headerLines[$lineCount++];
581:       }
582:       
583:       //Append token to the line
584:       if ("\r\n" != $token)

587:       }
588:     }
