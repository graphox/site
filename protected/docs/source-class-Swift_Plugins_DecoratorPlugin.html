 25: /**
 26:  * Allows customization of Messages on-the-fly.
 27:  * 
 28:  * @package Swift
 29:  * @subpackage Plugins
 30:  * 
 31:  * @author Chris Corbyn
 32:  */

 34:   implements Swift_Events_SendListener, Swift_Plugins_Decorator_Replacements
 35: {
 36: 
 37:   /** The replacement map */
 38:   private $_replacements;
 39:   
 40:   /** The body as it was before replacements */
 41:   private $_orginalBody;
 42: 
 43:   /** The original subject of the message, before replacements */
 44:   private $_originalSubject;
 45: 
 46:   /** Bodies of children before they are replaced */
 47:   private $_originalChildBodies = array();
 48: 
 49:   /** The Message that was last replaced */
 50:   private $_lastMessage;
 51: 
 52:   /**
 53:    * Create a new DecoratorPlugin with $replacements.
 54:    * 
 55:    * The $replacements can either be an associative array, or an implementation
 56:    * of {@link Swift_Plugins_Decorator_Replacements}.
 57:    * 
 58:    * When using an array, it should be of the form:
 59:    * <code>
 60:    * $replacements = array(
 61:    *  "address1@domain.tld" => array("{a}" => "b", "{c}" => "d"),
 62:    *  "address2@domain.tld" => array("{a}" => "x", "{c}" => "y")
 63:    * )
 64:    * </code>
 65:    * 
 66:    * When using an instance of {@link Swift_Plugins_Decorator_Replacements},
 67:    * the object should return just the array of replacements for the address
 68:    * given to {@link Swift_Plugins_Decorator_Replacements::getReplacementsFor()}.
 69:    * 
 70:    * @param mixed $replacements
 71:    */
 72:   public function __construct($replacements)
 73:   {
 74:     if (!($replacements instanceof Swift_Plugins_Decorator_Replacements))
 75:     {
 76:       $this->_replacements = (array) $replacements;
 77:     }
 78:     else
 79:     {
 80:       $this->_replacements = $replacements;
 81:     }
 82:   }
 83: 
 84:   /**
 85:    * Invoked immediately before the Message is sent.
 86:    * 
 87:    * @param Swift_Events_SendEvent $evt
 88:    */
 89:   public function beforeSendPerformed(Swift_Events_SendEvent $evt)
 90:   {
 91:     $message = $evt->getMessage();
 92:     $this->_restoreMessage($message);
 93:     $to = array_keys($message->getTo());
 94:     $address = array_shift($to);
 95:     if ($replacements = $this->getReplacementsFor($address))
 96:     {
 97:       $body = $message->getBody();
 98:       $search = array_keys($replacements);
 99:       $replace = array_values($replacements);
100:       $bodyReplaced = str_replace(
101:         $search, $replace, $body
102:         );
103:       if ($body != $bodyReplaced)
104:       {
105:         $this->_originalBody = $body;
106:         $message->setBody($bodyReplaced);
107:       }
108:       $subject = $message->getSubject();
109:       $subjectReplaced = str_replace(
110:         $search, $replace, $subject
111:         );
112:       if ($subject != $subjectReplaced)
113:       {
114:         $this->_originalSubject = $subject;
115:         $message->setSubject($subjectReplaced);
116:       }
117:       $children = (array) $message->getChildren();
118:       foreach ($children as $child)
119:       {
120:         list($type, ) = sscanf($child->getContentType(), '%[^/]/%s');
121:         if ('text' == $type)
122:         {
123:           $body = $child->getBody();
124:           $bodyReplaced = str_replace(
125:             $search, $replace, $body
126:             );
127:           if ($body != $bodyReplaced)
128:           {
129:             $child->setBody($bodyReplaced);
130:             $this->_originalChildBodies[$child->getId()] = $body;
131:           }
132:         }
133:       }
134:       $this->_lastMessage = $message;
135:     }
136:   }
137:   
138:   /**
139:    * Find a map of replacements for the address.
140:    * 
141:    * If this plugin was provided with a delegate instance of
142:    * {@link Swift_Plugins_Decorator_Replacements} then the call will be
143:    * delegated to it.  Otherwise, it will attempt to find the replacements
144:    * from the array provided in the constructor.
145:    * 
146:    * If no replacements can be found, an empty value (NULL) is returned.
147:    * 
148:    * @param string $address
149:    * 
150:    * @return array
151:    */
152:   public function getReplacementsFor($address)
153:   {
154:     if ($this->_replacements instanceof Swift_Plugins_Decorator_Replacements)
155:     {
156:       return $this->_replacements->getReplacementsFor($address);
157:     }
158:     else
159:     {
160:       return isset($this->_replacements[$address])
161:         ? $this->_replacements[$address]
162:         : null
163:         ;
164:     }
165:   }
166: 
167:   /**
168:    * Invoked immediately after the Message is sent.
169:    * 
170:    * @param Swift_Events_SendEvent $evt
171:    */
172:   public function sendPerformed(Swift_Events_SendEvent $evt)
173:   {
174:     $this->_restoreMessage($evt->getMessage());
175:   }

179:   /** Restore a changed message back to its original state */
180:   private function _restoreMessage(Swift_Mime_Message $message)
181:   {
182:     if ($this->_lastMessage === $message)
183:     {
184:       if (isset($this->_originalBody))
185:       {
186:         $message->setBody($this->_originalBody);
187:         $this->_originalBody = null;
188:       }
189:       if (isset($this->_originalSubject))
190:       {
191:         $message->setSubject($this->_originalSubject);
192:         $this->_originalSubject = null;
193:       }
194:       if (!empty($this->_originalChildBodies))
195:       {
196:         $children = (array) $message->getChildren();
197:         foreach ($children as $child)
198:         {
199:           $id = $child->getId();
200:           if (array_key_exists($id, $this->_originalChildBodies))
201:           {
202:             $child->setBody($this->_originalChildBodies[$id]);
203:           }
204:         }
205:         $this->_originalChildBodies = array();
206:       }
207:       $this->_lastMessage = null;
208:     }
209:   }
