163:     /**

176:             if (isset($item['visible']) && !$item['visible'])

180:             }

182:             if (!is_array($item)) {
183:                 continue;

190:                 $items[$i]['label'] = CHtml::encode($item['label']);
191: 
192:             if (($this->encodeLabel && !isset($item['encodeLabel']))
193:                     || (isset($item['encodeLabel']) && $item['encodeLabel'] !== false))
194:                 $items[$i]['label'] = CHtml::encode($item['label']);
