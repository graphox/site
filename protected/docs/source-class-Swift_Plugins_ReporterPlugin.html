26:  * Does real time reporting of pass/fail for each recipient.
27:  * @package Swift
28:  * @subpackage Plugins
29:  * @author Chris Corbyn
30:  */
31: class Swift_Plugins_ReporterPlugin
32:   implements Swift_Events_SendListener
33: {
34:   
35:   /**
36:    * The reporter backend which takes notifications.
37:    * @var Swift_Plugin_Reporter
38:    * @access private
39:    */
40:   private $_reporter;
41:   
42:   /**
43:    * Create a new ReporterPlugin using $reporter.
44:    * @param Swift_Plugins_Reporter $reporter
45:    */
46:   public function __construct(Swift_Plugins_Reporter $reporter)
47:   {
48:     $this->_reporter = $reporter;
49:   }
50:   
51:   /**
52:    * Not used.
53:    */
54:   public function beforeSendPerformed(Swift_Events_SendEvent $evt)
55:   {
56:   }
57:   
58:   /**
59:    * Invoked immediately after the Message is sent.
60:    * @param Swift_Events_SendEvent $evt
61:    */
62:   public function sendPerformed(Swift_Events_SendEvent $evt)
63:   {
64:     $message = $evt->getMessage();
65:     $failures = array_flip($evt->getFailedRecipients());
66:     foreach ((array) $message->getTo() as $address => $null)
67:     {
68:       $this->_reporter->notify(
69:         $message, $address, (array_key_exists($address, $failures)
70:         ? Swift_Plugins_Reporter::RESULT_FAIL
71:         : Swift_Plugins_Reporter::RESULT_PASS)
72:         );
73:     }
74:     foreach ((array) $message->getCc() as $address => $null)
75:     {
76:       $this->_reporter->notify(
77:         $message, $address, (array_key_exists($address, $failures)
78:         ? Swift_Plugins_Reporter::RESULT_FAIL
79:         : Swift_Plugins_Reporter::RESULT_PASS)
80:         );
81:     }
82:     foreach ((array) $message->getBcc() as $address => $null)
83:     {
84:       $this->_reporter->notify(
85:         $message, $address, (array_key_exists($address, $failures)
86:         ? Swift_Plugins_Reporter::RESULT_FAIL
87:         : Swift_Plugins_Reporter::RESULT_PASS)
88:         );
89:     }
90:   }
