 57:    */

 62:    * @param Swift_StreamFilter $filter
 63:    * @param string $key
 64:    */
 65:   public function addFilter(Swift_StreamFilter $filter, $key)
 66:   {
 67:     $this->_filters[$key] = $filter;
 68:   }

 71:    * Remove an already present StreamFilter based on its $key.
 72:    * @param string $key
 73:    */
 74:   public function removeFilter($key)
 75:   {
 76:     unset($this->_filters[$key]);
 77:   }
 78:   
 79:   /**
 80:    * Writes $bytes to the end of the stream.
 81:    * @param string $bytes
 82:    * @throws Swift_IoException
 83:    */
 84:   public function write($bytes)
 85:   {
 86:     $this->_writeBuffer .= $bytes;
 87:     foreach ($this->_filters as $filter)
 88:     {
 89:       if ($filter->shouldBuffer($this->_writeBuffer))
 90:       {
 91:         return;
 92:       }
 93:     }
 94:     $this->_doWrite($this->_writeBuffer);
 95:     return ++$this->_sequence;
 96:   }
 97:   
 98:   /**
 99:    * For any bytes that are currently buffered inside the stream, force them
100:    * off the buffer.
101:    * 
102:    * @throws Swift_IoException
103:    */
104:   public function commit()
105:   {
106:     $this->_doWrite($this->_writeBuffer);
107:   }
108:   
109:   /**
110:    * Attach $is to this stream.
111:    * The stream acts as an observer, receiving all data that is written.
112:    * All {@link write()} and {@link flushBuffers()} operations will be mirrored.
113:    * 
114:    * @param Swift_InputByteStream $is
115:    */
116:   public function bind(Swift_InputByteStream $is)
117:   {
118:     $this->_mirrors[] = $is;
119:   }
120:   
121:   /**
122:    * Remove an already bound stream.
123:    * If $is is not bound, no errors will be raised.
124:    * If the stream currently has any buffered data it will be written to $is
125:    * before unbinding occurs.
126:    * 
127:    * @param Swift_InputByteStream $is
128:    */
129:   public function unbind(Swift_InputByteStream $is)
130:   {
131:     foreach ($this->_mirrors as $k => $stream)
132:     {
133:       if ($is === $stream)
134:       {
135:         if ($this->_writeBuffer !== '')
136:         {
137:           $stream->write($this->_filter($this->_writeBuffer));
138:         }
139:         unset($this->_mirrors[$k]);
140:       }
141:     }
142:   }
143:   
144:   /**
145:    * Flush the contents of the stream (empty it) and set the internal pointer
146:    * to the beginning.
147:    * @throws Swift_IoException
148:    */
149:   public function flushBuffers()
150:   {
151:     if ($this->_writeBuffer !== '')
152:     {
153:       $this->_doWrite($this->_writeBuffer);
154:     }
155:     $this->_flush();
156:     
157:     foreach ($this->_mirrors as $stream)
158:     {
159:       $stream->flushBuffers();
160:     }
161:   }
162:   
163:   // -- Private methods
164:   
165:   /** Run $bytes through all filters */
166:   private function _filter($bytes)
167:   {
168:     foreach ($this->_filters as $filter)
169:     {
170:       $bytes = $filter->filter($bytes);
171:     }
172:     return $bytes;
173:   }
174:   
175:   /** Just write the bytes to the stream */
176:   private function _doWrite($bytes)
177:   {
178:     $this->_commit($this->_filter($bytes));
179:     
180:     foreach ($this->_mirrors as $stream)
181:     {
182:       $stream->write($bytes);
183:     }
184:     
185:     $this->_writeBuffer = '';
186:   }
