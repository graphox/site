262:   /**
263:    * Set all children of this entity.
264:    * @param array $children Swiift_Mime_Entity instances
265:    * @param int $compoundLevel For internal use only
266:    */
267:   public function setChildren(array $children, $compoundLevel = null)
268:   {
269:     //TODO: Try to refactor this logic
270:     
271:     $compoundLevel = isset($compoundLevel)
272:       ? $compoundLevel
273:       : $this->_getCompoundLevel($children)
274:       ;
275:     
276:     $immediateChildren = array();
277:     $grandchildren = array();
278:     $newContentType = $this->_userContentType;
279:     
280:     foreach ($children as $child)
281:     {
282:       $level = $this->_getNeededChildLevel($child, $compoundLevel);
283:       if (empty($immediateChildren)) //first iteration
284:       {
285:         $immediateChildren = array($child);
286:       }
287:       else
288:       {
289:         $nextLevel = $this->_getNeededChildLevel($immediateChildren[0], $compoundLevel);
290:         if ($nextLevel == $level)
291:         {
292:           $immediateChildren[] = $child;
293:         }
294:         elseif ($level < $nextLevel)
295:         {
296:           //Re-assign immediateChildren to grandchilden
297:           $grandchildren = array_merge($grandchildren, $immediateChildren);
298:           //Set new children
299:           $immediateChildren = array($child);
300:         }
301:         else
302:         {
303:           $grandchildren[] = $child;
304:         }
305:       }
306:     }
307:     
308:     if (!empty($immediateChildren))
309:     {
310:       $lowestLevel = $this->_getNeededChildLevel($immediateChildren[0], $compoundLevel);
311:       
312:       //Determine which composite media type is needed to accomodate the
313:       // immediate children
314:       foreach ($this->_compositeRanges as $mediaType => $range)
315:       {
316:         if ($lowestLevel > $range[0]
317:           && $lowestLevel <= $range[1])
318:         {
319:           $newContentType = $mediaType;
320:           break;
321:         }
322:       }
323:       
324:       //Put any grandchildren in a subpart
325:       if (!empty($grandchildren))
326:       {
327:         $subentity = $this->_createChild();
328:         $subentity->_setNestingLevel($lowestLevel);
329:         $subentity->setChildren($grandchildren, $compoundLevel);
330:         array_unshift($immediateChildren, $subentity);
331:       }
332:     }
333:     
334:     $this->_immediateChildren = $immediateChildren;
335:     $this->_children = $children;
336:     $this->_setContentTypeInHeaders($newContentType);
337:     $this->_fixHeaders();
338:     $this->_sortChildren();
339:     
340:     return $this;
341:   }

447:    * Get this entire entity as a string.
448:    * @return string
449:    */
450:   public function toString()
451:   {
452:     $string = $this->_headers->toString();
453:     if (isset($this->_body) && empty($this->_immediateChildren))
454:     {
455:       if ($this->_cache->hasKey($this->_cacheKey, 'body'))
456:       {
457:         $body = $this->_cache->getString($this->_cacheKey, 'body');
458:       }
459:       else
460:       {
461:         $body = "\r\n" . $this->_encoder->encodeString($this->getBody(), 0,
462:           $this->getMaxLineLength()
463:           );
464:         $this->_cache->setString($this->_cacheKey, 'body', $body,
465:           Swift_KeyCache::MODE_WRITE
466:           );
467:       }
468:       $string .= $body;
469:     }
470:     
471:     if (!empty($this->_immediateChildren))
472:     {
473:       foreach ($this->_immediateChildren as $child)
474:       {
475:         $string .= "\r\n\r\n--" . $this->getBoundary() . "\r\n";
476:         $string .= $child->toString();
477:       }
478:       $string .= "\r\n\r\n--" . $this->getBoundary() . "--\r\n";
479:     }
480:     
481:     return $string;
482:   }

484:   /**
485:    * Write this entire entity to a {@link Swift_InputByteStream}.
486:    * @param Swift_InputByteStream
487:    */
488:   public function toByteStream(Swift_InputByteStream $is)
489:   {
490:     $is->write($this->_headers->toString());
491:     $is->commit();
492:     
493:     if (empty($this->_immediateChildren))
494:     {
495:       if (isset($this->_body))
496:       {
497:         if ($this->_cache->hasKey($this->_cacheKey, 'body'))
498:         {
499:           $this->_cache->exportToByteStream($this->_cacheKey, 'body', $is);
500:         }
501:         else
502:         {
503:           $cacheIs = $this->_cache->getInputByteStream($this->_cacheKey, 'body');
504:           if ($cacheIs)
505:           {
506:             $is->bind($cacheIs);
507:           }
508:           
509:           $is->write("\r\n");
510:           
511:           if ($this->_body instanceof Swift_OutputByteStream)
512:           {
513:             $this->_body->setReadPointer(0);
514:             
515:             $this->_encoder->encodeByteStream($this->_body, $is, 0,
516:               $this->getMaxLineLength()
517:               );
518:           }
519:           else
520:           {
521:             $is->write($this->_encoder->encodeString(
522:               $this->getBody(), 0, $this->getMaxLineLength()
523:               ));
524:           }
525:           
526:           if ($cacheIs)
527:           {
528:             $is->unbind($cacheIs);
529:           }
530:         }
531:       }
532:     }
533:     
534:     if (!empty($this->_immediateChildren))
535:     {
536:       foreach ($this->_immediateChildren as $child)
537:       {
538:         $is->write("\r\n\r\n--" . $this->getBoundary() . "\r\n");
539:         $child->toByteStream($is);
540:       }
541:       $is->write("\r\n\r\n--" . $this->getBoundary() . "--\r\n");
542:     }
543:   }

686:   private function _getCompoundLevel($children)
687:   {
688:     $level = 0;
689:     foreach ($children as $child)

691:       $level |= $child->getNestingLevel();
692:     }

695:   
696:   private function _getNeededChildLevel($child, $compoundLevel)
697:   {
698:     $filter = array();
699:     foreach ($this->_compoundLevelFilters as $bitmask => $rules)
700:     {
701:       if (($compoundLevel & $bitmask) === $bitmask)
702:       {
703:         $filter = $rules + $filter;
704:       }
705:     }
