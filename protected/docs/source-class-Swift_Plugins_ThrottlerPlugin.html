 26: /**
 27:  * Throttles the rate at which emails are sent.
 28:  * @package Swift
 29:  * @subpackage Plugins
 30:  * @author Chris Corbyn
 31:  */
 32: class Swift_Plugins_ThrottlerPlugin
 33:   extends Swift_Plugins_BandwidthMonitorPlugin
 34:   implements Swift_Plugins_Sleeper, Swift_Plugins_Timer
 35: {
 36:   
 37:   /** Flag for throttling in bytes per minute */
 38:   const BYTES_PER_MINUTE = 0x01;
 39:   
 40:   /** Flag for throttling in emails per minute */
 41:   const MESSAGES_PER_MINUTE = 0x10;
 42:   
 43:   /**
 44:    * The Sleeper instance for sleeping.
 45:    * @var Swift_Plugins_Sleeper
 46:    * @access private
 47:    */
 48:   private $_sleeper;
 49:   
 50:   /**
 51:    * The Timer instance which provides the timestamp.
 52:    * @var Swift_Plugins_Timer
 53:    * @access private
 54:    */
 55:   private $_timer;
 56:   
 57:   /**
 58:    * The time at which the first email was sent.
 59:    * @var int
 60:    * @access private
 61:    */
 62:   private $_start;
 63:   
 64:   /**
 65:    * The rate at which messages should be sent.
 66:    * @var int
 67:    * @access private
 68:    */
 69:   private $_rate;
 70:   
 71:   /**
 72:    * The mode for throttling.
 73:    * This is {@link BYTES_PER_MINUTE} or {@link MESSAGES_PER_MINUTE}
 74:    * @var int
 75:    * @access private
 76:    */
 77:   private $_mode;
 78:   
 79:   /**
 80:    * An internal counter of the number of messages sent.
 81:    * @var int
 82:    * @access private
 83:    */
 84:   private $_messages = 0;
 85:   
 86:   /**
 87:    * Create a new ThrottlerPlugin.
 88:    * @param int $rate
 89:    * @param int $mode, defaults to {@link BYTES_PER_MINUTE}
 90:    * @param Swift_Plugins_Sleeper $sleeper (only needed in testing)
 91:    * @param Swift_Plugins_Timer $timer (only needed in testing)
 92:    */
 93:   public function __construct($rate, $mode = self::BYTES_PER_MINUTE,
 94:     Swift_Plugins_Sleeper $sleeper = null, Swift_Plugins_Timer $timer = null)
 95:   {
 96:     $this->_rate = $rate;
 97:     $this->_mode = $mode;
 98:     $this->_sleeper = $sleeper;
 99:     $this->_timer = $timer;
100:   }
101:   
102:   /**
103:    * Invoked immediately before the Message is sent.
104:    * @param Swift_Events_SendEvent $evt
105:    */
106:   public function beforeSendPerformed(Swift_Events_SendEvent $evt)
107:   {
108:     $time = $this->getTimestamp();
109:     if (!isset($this->_start))
110:     {
111:       $this->_start = $time;
112:     }
113:     $duration = $time - $this->_start;
114:     
115:     if (self::BYTES_PER_MINUTE == $this->_mode)
116:     {
117:       $sleep = $this->_throttleBytesPerMinute($duration);
118:     }
119:     else
120:     {
121:       $sleep = $this->_throttleMessagesPerMinute($duration);
122:     }
123:     
124:     if ($sleep > 0)
125:     {
126:       $this->sleep($sleep);
127:     }
128:   }
129:   
130:   /**
131:    * Invoked when a Message is sent.
132:    * @param Swift_Events_SendEvent $evt
133:    */
134:   public function sendPerformed(Swift_Events_SendEvent $evt)
135:   {
136:     parent::sendPerformed($evt);
137:     ++$this->_messages;
138:   }
139:   
140:   /**
141:    * Sleep for $seconds.
142:    * @param int $seconds
143:    */
144:   public function sleep($seconds)
145:   {
146:     if (isset($this->_sleeper))
147:     {
148:       $this->_sleeper->sleep($seconds);
149:     }
150:     else
151:     {
152:       sleep($seconds);
153:     }
154:   }
155:   
156:   /**
157:    * Get the current UNIX timestamp
158:    * @return int
159:    */
160:   public function getTimestamp()
161:   {
162:     if (isset($this->_timer))
163:     {
164:       return $this->_timer->getTimestamp();
165:     }
166:     else
167:     {
168:       return time();
169:     }
170:   }
171:   
172:   // -- Private methods
173:   
174:   /**
175:    * Get a number of seconds to sleep for.
176:    * @param int $timePassed
177:    * @return int
178:    * @access private
179:    */
180:   private function _throttleBytesPerMinute($timePassed)
181:   {
182:     $expectedDuration = $this->getBytesOut() / ($this->_rate / 60);
183:     return (int) ceil($expectedDuration - $timePassed);
184:   }
185:   
186:   /**
187:    * Get a number of seconds to sleep for.
188:    * @param int $timePassed
189:    * @return int
190:    * @access private
191:    */
192:   private function _throttleMessagesPerMinute($timePassed)
193:   {
194:     $expectedDuration = $this->_messages / ($this->_rate / 60);
195:     return (int) ceil($expectedDuration - $timePassed);
196:   }
