25: /**
26:  * Handles CRAM-MD5 authentication.
27:  * @package Swift
28:  * @subpackage Transport
29:  * @author Chris Corbyn
30:  */
31: class Swift_Transport_Esmtp_Auth_CramMd5Authenticator
32:   implements Swift_Transport_Esmtp_Authenticator
33: {
34:   
35:   /**
36:    * Get the name of the AUTH mechanism this Authenticator handles.
37:    * @return string
38:    */
39:   public function getAuthKeyword()
40:   {
41:     return 'CRAM-MD5';
42:   }
43:   
44:   /**
45:    * Try to authenticate the user with $username and $password.
46:    * @param Swift_Transport_SmtpAgent $agent
47:    * @param string $username
48:    * @param string $password
49:    * @return boolean
50:    */
51:   public function authenticate(Swift_Transport_SmtpAgent $agent,
52:     $username, $password)
53:   {
54:     try
55:     {
56:       $challenge = $agent->executeCommand("AUTH CRAM-MD5\r\n", array(334));
57:       $challenge = base64_decode(substr($challenge, 4));
58:       $message = base64_encode(
59:         $username . ' ' . $this->_getResponse($password, $challenge)
60:         );
61:       $agent->executeCommand(sprintf("%s\r\n", $message), array(235));
62:       return true;
63:     }
64:     catch (Swift_TransportException $e)
65:     {
66:       $agent->executeCommand("RSET\r\n", array(250));
67:       return false;
68:     }
69:   }
70:   
71:   /**
72:    * Generate a CRAM-MD5 response from a server challenge.
73:    * @param string $secret
74:    * @param string $challenge
75:    * @return string
76:    */
77:   private function _getResponse($secret, $challenge)
78:   {
79:     if (strlen($secret) > 64)
80:     {
81:       $secret = pack('H32', md5($secret));
82:     }
83:     
84:     if (strlen($secret) < 64)
85:     {
86:       $secret = str_pad($secret, 64, chr(0));
87:     }
88:     
89:     $k_ipad = substr($secret, 0, 64) ^ str_repeat(chr(0x36), 64);
90:     $k_opad = substr($secret, 0, 64) ^ str_repeat(chr(0x5C), 64);
91: 
92:     $inner  = pack('H32', md5($k_ipad . $challenge));
93:     $digest = md5($k_opad . $inner);
94: 
95:     return $digest;
96:   }
