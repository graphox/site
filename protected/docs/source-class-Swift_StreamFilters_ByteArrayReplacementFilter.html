 23: /**
 24:  * Processes bytes as they pass through a buffer and replaces sequences in it.
 25:  * This stream filter deals with Byte arrays rather than simple strings.
 26:  * @package Swift
 27:  * @author Chris Corbyn
 28:  */
 29: class Swift_StreamFilters_ByteArrayReplacementFilter
 30:   implements Swift_StreamFilter
 31: {
 32: 
 33:   /** The needle(s) to search for */
 34:   private $_search;
 35: 
 36:   /** The replacement(s) to make */
 37:   private $_replace;
 38: 
 39:   /** The Index for searching */
 40:   private $_index;
 41: 
 42:   /** The Search Tree */
 43:   private $_tree = array();
 44: 
 45:   /**  Gives the size of the largest search */
 46:   private $_treeMaxLen = 0;
 47:   
 48:   private $_repSize;
 49: 
 50:   /**
 51:    * Create a new ByteArrayReplacementFilter with $search and $replace.
 52:    * @param array $search
 53:    * @param array $replace
 54:    */
 55:   public function __construct($search, $replace)
 56:   {
 57:     $this->_search = $search;
 58:     $this->_index = array();
 59:     $this->_tree = array();
 60:     $this->_replace = array();
 61:     $this->_repSize = array();
 62:     
 63:     $tree = null;
 64:     $i = null;
 65:     $last_size = $size = 0;
 66:     foreach ($search as $i => $search_element)
 67:     {
 68:       if ($tree !== null)
 69:       {
 70:         $tree[-1] = min (count($replace) - 1, $i - 1);
 71:         $tree[-2] = $last_size;
 72:       }
 73:       $tree = &$this->_tree;
 74:       if (is_array ($search_element))
 75:       {
 76:         foreach ($search_element as $k => $char)
 77:         {
 78:           $this->_index[$char] = true;
 79:           if (!isset($tree[$char]))
 80:           {
 81:             $tree[$char] = array();
 82:           }
 83:           $tree = &$tree[$char];
 84:         }
 85:         $last_size = $k+1;
 86:         $size = max($size, $last_size);
 87:       }
 88:       else
 89:       {
 90:         $last_size = 1;
 91:         if (!isset($tree[$search_element]))
 92:         {
 93:           $tree[$search_element] = array();
 94:         }
 95:         $tree = &$tree[$search_element];
 96:         $size = max($last_size, $size);
 97:         $this->_index[$search_element] = true;
 98:       }
 99:     }
100:     if ($i !== null)
101:     {
102:       $tree[-1] = min (count ($replace) - 1, $i);
103:       $tree[-2] = $last_size;
104:       $this->_treeMaxLen = $size;
105:     }
106:     foreach ($replace as $rep)
107:     {
108:       if (!is_array($rep))
109:       {
110:         $rep = array ($rep);
111:       }
112:       $this->_replace[] = $rep;
113:     }
114:     for ($i = count($this->_replace) - 1; $i >= 0; --$i)
115:     {
116:       $this->_replace[$i] = $rep = $this->filter($this->_replace[$i], $i);
117:       $this->_repSize[$i] = count($rep);
118:     }
119:   }
120: 
121:   /**
122:    * Returns true if based on the buffer passed more bytes should be buffered.
123:    * @param array $buffer
124:    * @return boolean
125:    */
126:   public function shouldBuffer($buffer)
127:   {
128:     $endOfBuffer = end($buffer);
129:     return isset ($this->_index[$endOfBuffer]);
130:   }
131: 
132:   /**
133:    * Perform the actual replacements on $buffer and return the result.
134:    * @param array $buffer
135:    * @return array
136:    */
137:   public function filter($buffer, $_minReplaces = -1)
138:   {
139:     if ($this->_treeMaxLen == 0)
140:     {
141:       return $buffer;
142:     }
143:     
144:     $newBuffer = array();
145:     $buf_size = count($buffer);
146:     for ($i = 0; $i < $buf_size; ++$i)
147:     {
148:       $search_pos = $this->_tree;
149:       $last_found = PHP_INT_MAX;
150:       // We try to find if the next byte is part of a search pattern
151:       for ($j = 0; $j <= $this->_treeMaxLen; ++$j)
152:       {
153:         // We have a new byte for a search pattern
154:         if (isset ($buffer [$p = $i + $j]) && isset($search_pos[$buffer[$p]]))
155:         {
156:           $search_pos = $search_pos[$buffer[$p]];
157:           // We have a complete pattern, save, in case we don't find a better match later
158:           if (isset($search_pos[- 1]) && $search_pos[-1] < $last_found
159:             && $search_pos[-1] > $_minReplaces)
160:           {
161:             $last_found = $search_pos[-1];
162:             $last_size = $search_pos[-2];
163:           }
164:         }
165:         // We got a complete pattern
166:         elseif ($last_found !== PHP_INT_MAX)
167:         {
168:           // Adding replacement datas to output buffer
169:           $rep_size = $this->_repSize[$last_found];
170:           for ($j = 0; $j < $rep_size; ++$j)
171:           {
172:             $newBuffer[] = $this->_replace[$last_found][$j];
173:           }
174:           // We Move cursor forward
175:           $i += $last_size - 1;
176:           // Edge Case, last position in buffer
177:           if ($i >= $buf_size)
178:           {
179:             $newBuffer[] = $buffer[$i];
180:           }
181:           
182:           // We start the next loop
183:           continue 2;
184:         }
185:         else
186:         {
187:           // this byte is not in a pattern and we haven't found another pattern
188:           break;
189:         }
190:       }
191:       // Normal byte, move it to output buffer
192:       $newBuffer[] = $buffer[$i];
193:     }
194:     
195:     return $newBuffer;
196:   }
