 38:    * Creates a new QpContentEncoder for the given CharacterStream.
 39:    * @param Swift_CharacterStream $charStream to use for reading characters
 40:    * @param Swift_StreamFilter $filter if canonicalization should occur
 41:    */
 42:   public function __construct(Swift_CharacterStream $charStream,
 43:     Swift_StreamFilter $filter = null)
 44:   {
 45:     parent::__construct($charStream, $filter);
 46:   }

 48:   /**
 49:    * Encode stream $in to stream $out.
 50:    * QP encoded strings have a maximum line length of 76 characters.
 51:    * If the first line needs to be shorter, indicate the difference with
 52:    * $firstLineOffset.
 53:    * @param Swift_OutputByteStream $os output stream
 54:    * @param Swift_InputByteStream $is input stream
 55:    * @param int $firstLineOffset
 56:    * @param int $maxLineLength
 57:    */
 58:   public function encodeByteStream(
 59:     Swift_OutputByteStream $os, Swift_InputByteStream $is, $firstLineOffset = 0,
 60:     $maxLineLength = 0)
 61:   {
 62:     if ($maxLineLength > 76 || $maxLineLength <= 0)
 63:     {
 64:       $maxLineLength = 76;
 65:     }
 66: 
 67:     $thisLineLength = $maxLineLength - $firstLineOffset;
 68: 
 69:     $this->_charStream->flushContents();
 70:     $this->_charStream->importByteStream($os);
 71: 
 72:     $currentLine = '';
 73:     $prepend = '';
 74:     $size=$lineLen=0;
 75: 
 76:     while (false !== $bytes = $this->_nextSequence())
 77:     {
 78:       //If we're filtering the input
 79:       if (isset($this->_filter))
 80:       {
 81:         //If we can't filter because we need more bytes
 82:         while ($this->_filter->shouldBuffer($bytes))
 83:         {
 84:           //Then collect bytes into the buffer
 85:           if (false === $moreBytes = $this->_nextSequence(1))
 86:           {
 87:             break;
 88:           }
 89: 
 90:           foreach ($moreBytes as $b)
 91:           {
 92:             $bytes[] = $b;
 93:           }
 94:         }
 95:         //And filter them
 96:         $bytes = $this->_filter->filter($bytes);
 97:       }
 98: 
 99:       $enc = $this->_encodeByteSequence($bytes, $size);
100:       if ($currentLine && $lineLen+$size >= $thisLineLength)
101:       {
102:         $is->write($prepend . $this->_standardize($currentLine));
103:         $currentLine = '';
104:         $prepend = "=\r\n";
105:         $thisLineLength = $maxLineLength;
106:         $lineLen=0;
107:       }
108:       $lineLen+=$size;
109:       $currentLine .= $enc;
110:     }
111:     if (strlen($currentLine))
112:     {
113:       $is->write($prepend . $this->_standardize($currentLine));
114:     }
115:   }
