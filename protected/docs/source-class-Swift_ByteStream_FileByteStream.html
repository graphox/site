 26: /**
 27:  * Allows reading and writing of bytes to and from a file.
 28:  * @package Swift
 29:  * @subpackage ByteStream
 30:  * @author Chris Corbyn
 31:  */
 32: class Swift_ByteStream_FileByteStream
 33:   extends Swift_ByteStream_AbstractFilterableInputStream
 34:   implements Swift_FileStream
 35: {
 36:   
 37:   /** The internal pointer offset */
 38:   private $_offset = 0;
 39:   
 40:   /** The path to the file */
 41:   private $_path;
 42:   
 43:   /** The mode this file is opened in for writing */
 44:   private $_mode;
 45:   
 46:   /** A lazy-loaded resource handle for reading the file */
 47:   private $_reader;
 48:   
 49:   /** A lazy-loaded resource handle for writing the file */
 50:   private $_writer;
 51:   
 52:   /** If magic_quotes_runtime is on, this will be true */
 53:   private $_quotes = false;
 54:   
 55:   /**
 56:    * Create a new FileByteStream for $path.
 57:    * @param string $path
 58:    * @param string $writable if true
 59:    */
 60:   public function __construct($path, $writable = false)
 61:   {
 62:     $this->_path = $path;
 63:     $this->_mode = $writable ? 'w+b' : 'rb';
 64:     $this->_quotes = get_magic_quotes_runtime();
 65:   }
 66:   
 67:   /**
 68:    * Get the complete path to the file.
 69:    * @return string
 70:    */
 71:   public function getPath()
 72:   {
 73:     return $this->_path;
 74:   }
 75:   
 76:   /**
 77:    * Reads $length bytes from the stream into a string and moves the pointer
 78:    * through the stream by $length. If less bytes exist than are requested the
 79:    * remaining bytes are given instead. If no bytes are remaining at all, boolean
 80:    * false is returned.
 81:    * @param int $length
 82:    * @return string
 83:    * @throws Swift_IoException
 84:    */
 85:   public function read($length)
 86:   {
 87:     $fp = $this->_getReadHandle();
 88:     if (!feof($fp))
 89:     {
 90:       if ($this->_quotes)
 91:       {
 92:         set_magic_quotes_runtime(0);
 93:       }
 94:       $bytes = fread($fp, $length);
 95:       if ($this->_quotes)
 96:       {
 97:         set_magic_quotes_runtime(1);
 98:       }
 99:       $this->_offset = ftell($fp);
100:       return $bytes;
101:     }
102:     else
103:     {
104:       return false;
105:     }
106:   }
107:   
108:   /**
109:    * Move the internal read pointer to $byteOffset in the stream.
110:    * @param int $byteOffset
111:    * @return boolean
112:    */
113:   public function setReadPointer($byteOffset)
114:   {
115:     if (isset($this->_reader))
116:     {
117:       fseek($this->_reader, $byteOffset, SEEK_SET);
118:     }
119:     $this->_offset = $byteOffset;
120:   }
121:   
122:   // -- Private methods
123:   
124:   /** Just write the bytes to the file */
125:   protected function _commit($bytes)
126:   {
127:     fwrite($this->_getWriteHandle(), $bytes);
128:     $this->_resetReadHandle();
129:   }
130:   
131:   /** Not used */
132:   protected function _flush()
133:   {
134:   }
135:   
136:   /** Get the resource for reading */
137:   private function _getReadHandle()
138:   {
139:     if (!isset($this->_reader))
140:     {
141:       if (!$this->_reader = fopen($this->_path, 'rb'))
142:       {
143:         throw new Swift_IoException(
144:           'Unable to open file for reading [' . $this->_path . ']'
145:           );
146:       }
147:       fseek($this->_reader, $this->_offset, SEEK_SET);
148:     }
149:     return $this->_reader;
150:   }
151:   
152:   /** Get the resource for writing */
153:   private function _getWriteHandle()
154:   {
155:     if (!isset($this->_writer))
156:     {
157:       if (!$this->_writer = fopen($this->_path, $this->_mode))
158:       {
159:         throw new Swift_IoException(
160:           'Unable to open file for writing [' . $this->_path . ']'
161:           );
162:       }
163:     }
164:     return $this->_writer;
165:   }
166:   
167:   /** Force a reload of the resource for writing */
168:   private function _resetWriteHandle()
169:   {
170:     if (isset($this->_writer))
171:     {
172:       fclose($this->_writer);
173:       $this->_writer = null;
174:     }
175:   }
176:   
177:   /** Force a reload of the resource for reading */
178:   private function _resetReadHandle()
179:   {
180:     if (isset($this->_reader))
181:     {
182:       fclose($this->_reader);
183:       $this->_reader = null;
184:     }
185:   }
