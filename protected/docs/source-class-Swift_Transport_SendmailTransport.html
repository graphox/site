 26: /**
 27:  * SendmailTransport for sending mail through a sendmail/postfix (etc..) binary.
 28:  * 
 29:  * Supported modes are -bs and -t, with any additional flags desired.
 30:  * It is advised to use -bs mode since error reporting with -t mode is not
 31:  * possible.
 32:  * 
 33:  * @package Swift
 34:  * @subpackage Transport
 35:  * @author Chris Corbyn
 36:  */
 37: class Swift_Transport_SendmailTransport
 38:   extends Swift_Transport_AbstractSmtpTransport
 39: {
 40:   
 41:   /**
 42:    * Connection buffer parameters.
 43:    * @var array
 44:    * @access protected
 45:    */
 46:   private $_params = array(
 47:     'timeout' => 30,
 48:     'blocking' => 1,
 49:     'command' => '/usr/sbin/sendmail -bs',
 50:     'type' => Swift_Transport_IoBuffer::TYPE_PROCESS
 51:     );
 52:   
 53:   /**
 54:    * Create a new SendmailTransport with $buf for I/O.
 55:    * @param Swift_Transport_IoBuffer $buf
 56:    * @param Swift_Events_EventDispatcher $dispatcher
 57:    */
 58:   public function __construct(Swift_Transport_IoBuffer $buf,
 59:     Swift_Events_EventDispatcher $dispatcher)
 60:   {
 61:     parent::__construct($buf, $dispatcher);
 62:   }
 63:   
 64:   /**
 65:    * Start the standalone SMTP session if running in -bs mode.
 66:    */
 67:   public function start()
 68:   {
 69:     if (false !== strpos($this->getCommand(), ' -bs'))
 70:     {
 71:       parent::start();
 72:     }
 73:   }
 74:   
 75:   /**
 76:    * Set the command to invoke.
 77:    * If using -t mode you are strongly advised to include -oi or -i in the
 78:    * flags. For example: /usr/sbin/sendmail -oi -t
 79:    * Swift will append a -f<sender> flag if one is not present.
 80:    * The recommended mode is "-bs" since it is interactive and failure notifications
 81:    * are hence possible.
 82:    * @param string $command
 83:    */
 84:   public function setCommand($command)
 85:   {
 86:     $this->_params['command'] = $command;
 87:     return $this;
 88:   }
 89:   
 90:   /**
 91:    * Get the sendmail command which will be invoked.
 92:    * @return string
 93:    */
 94:   public function getCommand()
 95:   {
 96:     return $this->_params['command'];
 97:   }
 98:   
 99:   /**
100:    * Send the given Message.
101:    * Recipient/sender data will be retreived from the Message API.
102:    * The return value is the number of recipients who were accepted for delivery.
103:    * NOTE: If using 'sendmail -t' you will not be aware of any failures until
104:    * they bounce (i.e. send() will always return 100% success).
105:    * @param Swift_Mime_Message $message
106:    * @param string[] &$failedRecipients to collect failures by-reference
107:    * @return int
108:    */
109:   public function send(Swift_Mime_Message $message, &$failedRecipients = null)
110:   {
111:     $failedRecipients = (array) $failedRecipients;
112:     $command = $this->getCommand();
113:     $buffer = $this->getBuffer();
114:     
115:     if (false !== strpos($command, ' -t'))
116:     {
117:       if ($evt = $this->_eventDispatcher->createSendEvent($this, $message))
118:       {
119:         $this->_eventDispatcher->dispatchEvent($evt, 'beforeSendPerformed');
120:         if ($evt->bubbleCancelled())
121:         {
122:           return 0;
123:         }
124:       }
125:       
126:       if (false === strpos($command, ' -f'))
127:       {
128:         $command .= ' -f' . $this->_getReversePath($message);
129:       }
130:       
131:       $buffer->initialize(array_merge($this->_params, array('command' => $command)));
132:       
133:       if (false === strpos($command, ' -i') && false === strpos($command, ' -oi'))
134:       {
135:         $buffer->setWriteTranslations(array("\r\n" => "\n", "\n." => "\n.."));
136:       }
137:       else
138:       {
139:         $buffer->setWriteTranslations(array("\r\n"=>"\n"));
140:       }
141:       
142:       $count = count((array) $message->getTo())
143:         + count((array) $message->getCc())
144:         + count((array) $message->getBcc())
145:         ;
146:       $message->toByteStream($buffer);
147:       $buffer->flushBuffers();
148:       $buffer->setWriteTranslations(array());
149:       $buffer->terminate();
150:       
151:       if ($evt)
152:       {
153:         $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
154:         $evt->setFailedRecipients($failedRecipients);
155:         $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
156:       }
157:       
158:       $message->generateId();
159:     }
160:     elseif (false !== strpos($command, ' -bs'))
161:     {
162:       $count = parent::send($message, $failedRecipients);
163:     }
164:     else
165:     {
166:       $this->_throwException(new Swift_TransportException(
167:         'Unsupported sendmail command flags [' . $command . ']. ' .
168:         'Must be one of "-bs" or "-t" but can include additional flags.'
169:         ));
170:     }
171:     
172:     return $count;
173:   }
174:   
175:   // -- Protected methods
176:   
177:   /** Get the params to initialize the buffer */
178:   protected function _getBufferParams()
179:   {
180:     return $this->_params;
181:   }
182:   
183: }
