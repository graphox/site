285:     /**
286:      * Registers a Bootstrap JavaScript plugin.
287:      * @param string $name the name of the plugin
288:      * @param string $selector the CSS selector
289:      * @param array $options the plugin options
290:      * @param string $defaultSelector the default CSS selector
291:      * @since 0.9.8
292:      */
293:     protected function registerPlugin($name, $selector = null, $options = array(), $defaultSelector = null)
294:     {
295:         if (!isset($selector) && empty($options))
296:         {
297:             // Initialization from extension configuration.
298:             $config = isset($this->plugins[$name]) ? $this->plugins[$name] : array();
299: 
300:             if (isset($config['selector']))
301:                 $selector = $config['selector'];
302: 
303:             if (isset($config['options']))
304:                 $options = $config['options'];
305: 
306:             if (!isset($selector))
307:                 $selector = $defaultSelector;
308:         }
309: 
310:         if (isset($selector))
311:         {
312:             $key = __CLASS__.'.'.md5($name.$selector.serialize($options).$defaultSelector);
313:             $options = !empty($options) ? CJavaScript::encode($options) : '';
314:             Yii::app()->clientScript->registerScript($key, "jQuery('{$selector}').{$name}({$options});");
315:         }
316:     }
