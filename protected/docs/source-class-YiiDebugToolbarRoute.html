181:     {
182:         foreach ($this->ipFilters as $filter)
183:         {
184:             $filter = trim($filter);
185:             // normal or incomplete IPv4
186:             if (preg_match('/^[\d\.]*\*?$/', $filter)) {
187:                 $filter = rtrim($filter, '*');
188:                 if (strncmp($ip, $filter, strlen($filter)) === 0)
189:                 {
190:                     return true;
191:                 }
192:             }
193:             // CIDR
194:             else if (preg_match('/^([\d\.]+)\/(\d+)$/', $filter, $match))
195:             {
196:                 if (self::matchIpMask($ip, $match[1], $match[2]))
197:                 {
