 36:     
 37:     public $itoa64 = './0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
 38:     public $iteration_count_log2 = 8;
 39:     public $portable_hashes = false;
 40:     public $random_state = '';

 42:     public function setUserPassword(&$model)
 43:     {
 44:         $model->password = $this->HashPassword($model->password);

 67:     public function init()
 68:     {
 69:         if ($this->iteration_count_log2 < 4 || $this->iteration_count_log2 > 31)
 70:             $this->iteration_count_log2 = 8;
 71:             
 72:         $this->random_state .= microtime();
 73: 
 74:         if (function_exists('getmypid'))
 75:             $this->random_state .= getmypid();
 76:     }

 78:     public function get_random_bytes($count)
 79:     {
 80:         $output = '';
 81:         if (is_readable('/dev/urandom') &&
 82:             ($fh = @fopen('/dev/urandom', 'rb'))) {
 83:             $output = fread($fh, $count);
 84:             fclose($fh);
 85:         }
 86: 
 87:         if (strlen($output) < $count) {
 88:             $output = '';
 89:             for ($i = 0; $i < $count; $i += 16) {
 90:                 $this->random_state =
 91:                     md5(microtime() . $this->random_state);
 92:                 $output .=
 93:                     pack('H*', md5($this->random_state));
 94:             }
 95:             $output = substr($output, 0, $count);
 96:         }
 97: 
 98:         return $output;
 99:     }

101:     public function encode64($input, $count)
102:     {
103:         $output = '';
104:         $i = 0;
105:         do {
106:             $value = ord($input[$i++]);
107:             $output .= $this->itoa64[$value & 0x3f];
108:             if ($i < $count)
109:                 $value |= ord($input[$i]) << 8;
110:             $output .= $this->itoa64[($value >> 6) & 0x3f];
111:             if ($i++ >= $count)
112:                 break;
113:             if ($i < $count)
114:                 $value |= ord($input[$i]) << 16;
115:             $output .= $this->itoa64[($value >> 12) & 0x3f];
116:             if ($i++ >= $count)
117:                 break;
118:             $output .= $this->itoa64[($value >> 18) & 0x3f];
119:         } while ($i < $count);
120: 
121:         return $output;
122:     }

124:     public function gensalt_private($input)
125:     {
126:         $output = '$P$';
127:         $output .= $this->itoa64[min($this->iteration_count_log2 +
128:             ((PHP_VERSION >= '5') ? 5 : 3), 30)];
129:         $output .= $this->encode64($input, 6);
130: 
131:         return $output;
132:     }

134:     public function crypt_private($password, $setting)
135:     {
136:         $output = '*0';
137:         if (substr($setting, 0, 2) == $output)
138:             $output = '*1';
139: 
140:         $id = substr($setting, 0, 3);
141:         # We use "$P$", phpBB3 uses "$H$" for the same thing
142:         if ($id != '$P$' && $id != '$H$')
143:             return $output;
144: 
145:         $count_log2 = strpos($this->itoa64, $setting[3]);
146:         if ($count_log2 < 7 || $count_log2 > 30)
147:             return $output;
148: 
149:         $count = 1 << $count_log2;
150: 
151:         $salt = substr($setting, 4, 8);
152:         if (strlen($salt) != 8)
153:             return $output;
154: 
155:         # We're kind of forced to use MD5 here since it's the only
156:         # cryptographic primitive available in all versions of PHP
157:         # currently in use.  To implement our own low-level crypto
158:         # in PHP would result in much worse performance and
159:         # consequently in lower iteration counts and hashes that are
160:         # quicker to crack (by non-PHP code).
161:         if (PHP_VERSION >= '5') {
162:             $hash = md5($salt . $password, TRUE);
163:             do {
164:                 $hash = md5($hash . $password, TRUE);
165:             } while (--$count);
166:         } else {
167:             $hash = pack('H*', md5($salt . $password));
168:             do {
169:                 $hash = pack('H*', md5($hash . $password));
170:             } while (--$count);
171:         }
172: 
173:         $output = substr($setting, 0, 12);
174:         $output .= $this->encode64($hash, 16);
175: 
176:         return $output;
177:     }
178: 
179:     public function gensalt_extended($input)
180:     {
181:         $count_log2 = min($this->iteration_count_log2 + 8, 24);
182:         # This should be odd to not reveal weak DES keys, and the
183:         # maximum valid value is (2**24 - 1) which is odd anyway.
184:         $count = (1 << $count_log2) - 1;
185: 
186:         $output = '_';
187:         $output .= $this->itoa64[$count & 0x3f];
188:         $output .= $this->itoa64[($count >> 6) & 0x3f];
189:         $output .= $this->itoa64[($count >> 12) & 0x3f];
190:         $output .= $this->itoa64[($count >> 18) & 0x3f];
191: 
192:         $output .= $this->encode64($input, 3);
193: 
194:         return $output;
195:     }

197:     public function gensalt_blowfish($input)
198:     {
199:         # This one needs to use a different order of characters and a
200:         # different encoding scheme from the one in encode64() above.
201:         # We care because the last character in our encoded string will
202:         # only represent 2 bits.  While two known implementations of
203:         # bcrypt will happily accept and correct a salt string which
204:         # has the 4 unused bits set to non-zero, we do not want to take
205:         # chances and we also do not want to waste an additional byte
206:         # of entropy.
207:         $itoa64 = './ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
208: 
209:         $output = '$2a$';
210:         $output .= chr(ord('0') + $this->iteration_count_log2 / 10);
211:         $output .= chr(ord('0') + $this->iteration_count_log2 % 10);
212:         $output .= '$';
213: 
214:         $i = 0;
215:         do {
216:             $c1 = ord($input[$i++]);
217:             $output .= $itoa64[$c1 >> 2];
218:             $c1 = ($c1 & 0x03) << 4;
219:             if ($i >= 16) {
220:                 $output .= $itoa64[$c1];
221:                 break;
222:             }
223: 
224:             $c2 = ord($input[$i++]);
225:             $c1 |= $c2 >> 4;
226:             $output .= $itoa64[$c1];
227:             $c1 = ($c2 & 0x0f) << 2;
228: 
229:             $c2 = ord($input[$i++]);
230:             $c1 |= $c2 >> 6;
231:             $output .= $itoa64[$c1];
232:             $output .= $itoa64[$c2 & 0x3f];
233:         } while (1);
234: 
235:         return $output;
236:     }

238:     public function HashPassword($password)
239:     {
240:         $random = '';
241: 
242:         if (CRYPT_BLOWFISH == 1 && !$this->portable_hashes) {
243:             $random = $this->get_random_bytes(16);
244:             $hash =
245:                 crypt($password, $this->gensalt_blowfish($random));
246:             if (strlen($hash) == 60)
247:                 return $hash;
248:         }
249: 
250:         if (CRYPT_EXT_DES == 1 && !$this->portable_hashes) {
251:             if (strlen($random) < 3)
252:                 $random = $this->get_random_bytes(3);
253:             $hash =
254:                 crypt($password, $this->gensalt_extended($random));
255:             if (strlen($hash) == 20)
256:                 return $hash;
257:         }
258: 
259:         if (strlen($random) < 6)
260:             $random = $this->get_random_bytes(6);
261:         $hash =
262:             $this->crypt_private($password,
263:             $this->gensalt_private($random));
264:         if (strlen($hash) == 34)
265:             return $hash;
266: 
267:         # Returning '*' on error is safe here, but would _not_ be safe
268:         # in a crypt(3)-like function used _both_ for generating new
269:         # hashes and for validating passwords against existing hashes.
270:         return '*';
271:     }
272: 
273:     public function CheckPassword($password, $stored_hash)
274:     {
275:         $hash = $this->crypt_private($password, $stored_hash);
276:         if ($hash[0] == '*')
277:             $hash = crypt($password, $stored_hash);
278: 
279:         return $hash == $stored_hash;
280:     }
