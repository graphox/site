 25: /**
 26:  * Handles RFC 2231 specified Encoding in Swift Mailer.
 27:  * @package Swift
 28:  * @subpackage Encoder
 29:  * @author Chris Corbyn
 30:  */
 31: class Swift_Encoder_Rfc2231Encoder implements Swift_Encoder
 32: {
 33:   
 34:   /**
 35:    * A character stream to use when reading a string as characters instead of bytes.
 36:    * @var Swift_CharacterStream
 37:    * @access private
 38:    */
 39:   private $_charStream;
 40:   
 41:   /**
 42:    * Creates a new Rfc2231Encoder using the given character stream instance.
 43:    * @param Swift_CharacterStream
 44:    */
 45:   public function __construct(Swift_CharacterStream $charStream)
 46:   {
 47:     $this->_charStream = $charStream;
 48:   }
 49:   
 50:   /**
 51:    * Takes an unencoded string and produces a string encoded according to
 52:    * RFC 2231 from it.
 53:    * @param string $string to encode
 54:    * @param int $firstLineOffset
 55:    * @param int $maxLineLength, optional, 0 indicates the default of 75 bytes
 56:    * @return string
 57:    */
 58:   public function encodeString($string, $firstLineOffset = 0,
 59:     $maxLineLength = 0)
 60:   {
 61:     $lines = array(); $lineCount = 0;
 62:     $lines[] = '';
 63:     $currentLine =& $lines[$lineCount++];
 64:     
 65:     if (0 >= $maxLineLength)
 66:     {
 67:       $maxLineLength = 75;
 68:     }
 69:     
 70:     $this->_charStream->flushContents();
 71:     $this->_charStream->importString($string);
 72:     
 73:     $thisLineLength = $maxLineLength - $firstLineOffset;
 74:     
 75:     while (false !== $char = $this->_charStream->read(4))
 76:     {
 77:       $encodedChar = rawurlencode($char);
 78:       if (0 != strlen($currentLine)
 79:         && strlen($currentLine . $encodedChar) > $thisLineLength)
 80:       {
 81:         $lines[] = '';
 82:         $currentLine =& $lines[$lineCount++];
 83:         $thisLineLength = $maxLineLength;
 84:       }
 85:       $currentLine .= $encodedChar;
 86:     }
 87:     
 88:     return implode("\r\n", $lines);
 89:   }
 90:   
 91:   /**
 92:    * Updates the charset used.
 93:    * @param string $charset
 94:    */
 95:   public function charsetChanged($charset)
 96:   {
 97:     $this->_charStream->setCharacterSet($charset);
 98:   }
